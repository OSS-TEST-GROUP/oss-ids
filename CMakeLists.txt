cmake_minimum_required(VERSION 3.21)

project(dds_ids LANGUAGES CXX)

include(GNUInstallDirs)
include(CTest)

if(CMAKE_INSTALL_PREFIX_INITIALIZED_TO_DEFAULT)
    set(CMAKE_INSTALL_PREFIX "${CMAKE_SOURCE_DIR}" CACHE PATH "Install path prefix" FORCE)
endif()

if(NOT CMAKE_BUILD_TYPE AND NOT CMAKE_CONFIGURATION_TYPES)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "Build type" FORCE)
endif()

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS ON)
set(CMAKE_FIND_PACKAGE_PREFER_CONFIG ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")

# Prefer packages generated by Conan (CMakeDeps) in the build generators folder
list(PREPEND CMAKE_PREFIX_PATH "${CMAKE_BINARY_DIR}/generators")

find_package(Threads REQUIRED)
find_package(fastdds REQUIRED CONFIG)
find_package(fastcdr REQUIRED CONFIG)
find_package(cxxopts REQUIRED CONFIG)
find_package(fmt REQUIRED CONFIG)
find_package(spdlog REQUIRED CONFIG)
find_package(nlohmann_json REQUIRED CONFIG)
find_package(libdds-ids CONFIG REQUIRED)

if(BUILD_TESTING)
    find_package(GTest CONFIG REQUIRED)
endif()

set(DDS_IDS_RUNTIME_LIBS
    fastdds
    fastcdr
    Threads::Threads
)

if(UNIX AND NOT APPLE)
    list(APPEND DDS_IDS_RUNTIME_LIBS rt m)
endif()

set(DDS_IDS_APP_LIBS
    cxxopts::cxxopts
    fmt::fmt
)

set(DDS_IDS_TYPE_DIR "${CMAKE_BINARY_DIR}/idl/typefile")

list(PREPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_SOURCE_DIR}/cmake")

string(LENGTH "${CMAKE_SOURCE_DIR}/" SOURCE_PATH_SIZE)

if(NOT DEFINED LOG_ACTIVE_LEVEL OR "${LOG_ACTIVE_LEVEL}" STREQUAL "")
    set(LOG_ACTIVE_LEVEL 0)
endif()

configure_file(${CMAKE_SOURCE_DIR}/config.h.cmake ${CMAKE_BINARY_DIR}/config_${PROJECT_NAME}.h)
configure_file(${CMAKE_SOURCE_DIR}/policy_rule/config_dds_v0.9.json ${CMAKE_BINARY_DIR}/bin/config_dds_v0.9.json COPYONLY)
configure_file(${CMAKE_SOURCE_DIR}/policy_rule/config_dds_v1.0.json ${CMAKE_BINARY_DIR}/bin/config_dds_v1.0.json COPYONLY)
configure_file(${CMAKE_SOURCE_DIR}/SecurityClient/config/sample_runtime_config.json ${CMAKE_BINARY_DIR}/bin/sample_runtime_config.json COPYONLY)
configure_file(${CMAKE_SOURCE_DIR}/SecurityClient/config/sample_policy_config.json ${CMAKE_BINARY_DIR}/bin/sample_policy_config.json COPYONLY)

add_subdirectory(idl)
add_subdirectory(SecurityManager)
add_subdirectory(SecurityClient)

install(FILES
    policy_rule/config_dds_v0.9.json
    policy_rule/config_dds_v1.0.json
    DESTINATION ${CMAKE_INSTALL_BINDIR}
)
